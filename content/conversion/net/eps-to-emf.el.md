        * Λειτουργικά συστήματα: Microsoft Windows, Linux, MacOS
        * Περιβάλλοντα ανάπτυξης: Microsoft Visual Studio, Xamarin, MonoDevelop

    code: |
        ```csharp    
        // Φορτώστε το αρχείο προέλευσης EPS για μετατροπή
          var converter = new GroupDocs.Conversion.Converter("input.eps");
          // Προετοιμασία επιλογών μετατροπής για τη μορφή στόχου EMF
          var convertOptions = converter.GetPossibleConversions()["emf"].ConvertOptions;
          // Μετατροπή σε μορφή EMF
          converter.Convert("output.emf", convertOptions);
        ```
