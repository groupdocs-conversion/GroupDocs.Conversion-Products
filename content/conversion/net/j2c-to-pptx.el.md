    code: |

        ```csharp    
        // Φορτώστε το αρχείο προέλευσης J2C για μετατροπή
          var converter = new GroupDocs.Conversion.Converter("input.j2c");
          // Προετοιμασία επιλογών μετατροπής για τη μορφή στόχου PPTX

          converter.Convert("output.pptx", convertOptions);
        ```
