    code: |
        ```csharp    
        // Ji bo veguhertinê pelê çavkaniyê JPX bar bike
          var converter = new GroupDocs.Conversion.Converter("input.jpx");
          // Vebijarkên veguhertinê ji bo forma armancê amade bikin SVG

          converter.Convert("output.svg", convertOptions);
        ```
