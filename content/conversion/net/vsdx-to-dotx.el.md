    code: |

        // Φορτώστε το αρχείο προέλευσης VSDX για μετατροπή
          var converter = new GroupDocs.Conversion.Converter("input.vsdx");
          // Προετοιμασία επιλογών μετατροπής για τη μορφή στόχου DOTX

          converter.Convert("output.dotx", convertOptions);
